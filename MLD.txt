candidat(#identifiant : VARCHAR, mot_de_passe : VARCHAR, nom : VARCHAR, prenom : VARCHAR, e_mail : VARCHAR, cv=>CV) 
avec identifiant clé, (mot_de_passe, nom, prenom, e_mail) NOT NULL et cv UNIQUE NOT NULL.

cv(#IDCV : INTEGER ,titre : VARCHAR, langue : ENUM, statut : ENUM, date_crea : TIMESTAMP, date_modif : TIMESTAMP, info => info_complementaire)
On ajoute une clé étrangère car aucune clé potentielle, IDCV clé, et titre, langue, statut, date_crea, date_modif NOT NULL.

referant(#nom_entreprise : VARCHAR, nom : VARCHAR, prenom : VARCHAR, #candidat => candidat)
avec nom_entreprise clé, (nom, prenom) NOT NULL, et candidat UNIQUE et NOT NULL.

publication(#ISBN : INTEGER, titre : VARCHAR, date : DATETIME, contenu : VARCHAR, #candidat => candidat)
avec ISBN clé, (titre, date, contenu) NOT NULL, et candidat NOT NULL et UNIQUE.

web(#URL : VARCHAR, type = ENUM, #candidat =>candidat)
avec URL clé, type NOT NULL et candidat NOT NULL et UNIQUE.

telephone(#numero : INTEGER, type = ENUM}, #candidat => candidat}
avec numero clé, type NOT NULL, et candidat NOT NULL et UNIQUE.

experience(#nom_entreprise : VARCHAR, titre_poste : VARCHAR, date_debut : DATETIME, date_fin : DATETIME, fonction : VARCHAR, secteur_act : VARCHAR, #cv => CV)
avec nom_entreprise clé, titre_poste, date_debut, date_fin, fonction et secteur d'activité NOT NULL, et cv UNIQUE NOT NULL.

info_complementaire(#IDINFO, permis : ENUM, vehicule : ENUM)
avec IDINFO clé étrangère car aucune clé candidate.

formation(#titre : VARCHAR,#etablissement : VARCHAR, type : VARCHAR, date_debut : DATE, date_fin : DATE,  pays : VARCHAR, ville : VARCHAR) 
avec (titre, etablissement) clé, type, date_debut, date_fin, pays et ville NOT NULL.

competence(#nom : VARCHAR)
avec nom clé.

langue(#langue : VARCHAR) 
avec langue clé.

association(#nom : VARCHAR, duree : INTEGER, poste : VARCHAR) 
avec nom clé.

---transformation N:M---
vie_associative(#association => association, #candidat => candidat)
avec (association, candidat) clé.

niveau(#langue => langue, #cv => CV, niveau = VARCHAR)
avec (langue, cv) clé, et langue, cv, niveau NOT NULL. 

assoc_comp(#competence => competence, #cv => CV)
avec (competence, cv) clé.

assoc_formation(#formation => formation, #cv => CV)
avec (formation, cv) clé.
